    <button onclick="palin()">Submit</button>

  <script>
    function palin() {
      const hey = document.getElementById("hey");
      const inp1Elem = document.getElementById("inp1");
      const inp2 = document.getElementById("inp2");
      const inp3 = document.getElementById("inp3");

      const val = inp1Elem.value.trim();

      if (val === "" || isNaN(val)) {
        inp2.value = "Please enter a valid number.";
        hey.style.boxShadow = "0 0 20px red";
        inp2.style.color = "red";
        inp3.value = "";
        return;
      }

      const num = parseInt(val);
      const temp = num;
      let rev = 0;

      let copy = num;
      while (copy !== 0) {
        let rem = copy % 10;
        rev = rev * 10 + rem;
        copy = Math.floor(copy / 10);
      }

      inp3.value = rev;

      if (rev === temp) {
        inp2.value = `So ${temp} is a palindrome`;
        hey.style.boxShadow = "0 0 20px rgb(51, 222, 51)";
        inp2.style.color = "rgb(51, 222, 51)";
        inp3.style.color = "rgb(51, 222, 51)";
      } else {
        inp2.value = `So ${temp} is not a palindrome`;
        hey.style.boxShadow = "0 0 20px red";
        inp2.style.color = "red";
        inp3.style.color = "red";
      }
    }

    function reloadPage() {
      location.reload();
    }
    document.getElementById("inp1").addEventListener("keypress", function(e) {
      if (e.key === "Enter") {
        palin();
      }
    });
    window.onload = () => {
      document.getElementById("inp1").focus();
    };
  </script>
